# AMD_HACKATHON - After 8 App

# 1. Create project
npm create vite@latest after8 -- --template react
cd after8

# 2. Replace src/ with project files

# 3. Run locally
npm install
npm run dev

# 4. Run tests
node tests/run.js

# 5. Build for deployment
npm run build
# → drag dist/ folder to netlify.com/drop for instant live URL
